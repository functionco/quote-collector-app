import React, { useState, useEffect, useCallback } from 'react';
import { GoogleUser } from './types';
import { initGoogleClient, signIn, signOut } from './services/googleService';
import Dashboard from './components/Dashboard';
import LoginScreen from './components/LoginScreen';

const App: React.FC = () => {
  const [user, setUser] = useState<GoogleUser | null>(null);
  const [isGapiLoaded, setIsGapiLoaded] = useState<boolean>(false);
  const [error, setError] = useState<string | null>(null);

  const updateUser = useCallback((currentUser: any) => {
    if (currentUser && currentUser.isSignedIn()) {
      const profile = currentUser.getBasicProfile();
      setUser({
        name: profile.getName(),
        email: profile.getEmail(),
        imageUrl: profile.getImageUrl(),
      });
    } else {
      setUser(null);
    }
  }, []);
  
  useEffect(() => {
    // The gapi script is loaded asynchronously. We need to check if it's available on the window object.
    const checkGapiReady = () => {
      if (window.gapi && window.gapi.load) {
        initGoogleClient((authInstance) => {
          setIsGapiLoaded(true);
          updateUser(authInstance.currentUser.get());
          authInstance.currentUser.listen(updateUser);
        }, (errMessage) => {
          console.error("Error initializing Google Client:", errMessage);
          setError(String(errMessage));
        });
      } else {
        setTimeout(checkGapiReady, 100); // Check again in 100ms
      }
    };
    checkGapiReady();
  }, [updateUser]);

  const handleSignIn = async () => {
    await signIn();
  };

  const handleSignOut = () => {
    signOut();
  };
  
  if (error) {
    return (
      <div className="flex items-center justify-center min-h-screen bg-background">
        <div className="bg-white p-8 rounded-lg shadow-xl text-center max-w-md">
          <h1 className="text-2xl font-bold text-red-500 mb-4">Initialization Error</h1>
          <p className="text-slate-600">{error}</p>
          <p className="text-xs text-slate-500 mt-4">Please ensure you have enabled the required APIs in your Google Cloud project and configured your credentials correctly in the build environment.</p>
        </div>
      </div>
    );
  }

  if (!isGapiLoaded) {
    return (
      <div className="flex items-center justify-center min-h-screen bg-background">
        <div className="flex flex-col items-center space-y-3">
          <svg className="animate-spin h-8 w-8 text-accent-2" xmlns="http://www.w3.org/2000/svg" fill="none" viewBox="0 0 24 24">
            <circle className="opacity-25" cx="12" cy="12" r="10" stroke="currentColor" strokeWidth="4"></circle>
            <path className="opacity-75" fill="currentColor" d="M4 12a8 8 0 018-8V0C5.373 0 0 5.373 0 12h4zm2 5.291A7.962 7.962 0 014 12H0c0 3.042 1.135 5.824 3 7.938l3-2.647z"></path>
          </svg>
          <span className="text-xl font-medium text-slate-700">Initializing Google Services...</span>
        </div>
      </div>
    );
  }

  return (
    <div className="min-h-screen text-slate-800 bg-background">
      {user ? (
        <Dashboard user={user} onSignOut={handleSignOut} />
      ) : (
        <LoginScreen onSignIn={handleSignIn} />
      )}
    </div>
  );
};

export default App;
